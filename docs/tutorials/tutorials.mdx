## Multi-Tab Code Blocks

<Tabs>
  <TabItem value="node" label="Node.js">

<CodeBlockPanel title="send.js" language="js">
{`const dokaai = require("@dokaai/sdk");

dokaai.send({
  subscriberId: "user_123",
  workflowId: "welcome",
  payload: { name: "Ava" },
});`}
</CodeBlockPanel>

  </TabItem>
  <TabItem value="python" label="Python">

<CodeBlockPanel title="send.py" language="python">
{`from dokaai import client

client.send(
    subscriber_id="user_123",
    workflow_id="welcome",
    payload={"name": "Ava"}
)`}
</CodeBlockPanel>

  </TabItem>
  <TabItem value="curl" label="cURL">

<CodeBlockPanel title="send.sh" language="bash">
{`curl -X POST https://api.dokaai.com/v1/send \\
  -H "Authorization: Bearer $DOKAAI_API_KEY" \\
  -H "Content-Type: application/json" \\
  -d '{"subscriberId":"user_123","workflowId":"welcome","payload":{"name":"Ava"}}'`}
</CodeBlockPanel>

  </TabItem>
</Tabs>
